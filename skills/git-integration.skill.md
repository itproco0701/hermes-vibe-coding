## Pre-Task: Safety Checkpoint

```bash
git_checkpoint_start() {
  local task_slug="$1"  # short description, used in branch name

  # 1. Verify git repo
  if ! git rev-parse --git-dir &>/dev/null; then
    echo "NOT_GIT_REPO"
    return 1
  fi

  # 2. Save current branch for rollback reference
  BASE_BRANCH=$(git branch --show-current)
  echo "Base branch: $BASE_BRANCH"

  # 3. Stash any dirty state
  if [[ -n "$(git status --porcelain)" ]]; then
    STASH_MSG="vibe-coding: pre-task stash $(date +%Y%m%d-%H%M%S)"
    git stash push --include-untracked -m "$STASH_MSG"
    echo "Stashed dirty state: $STASH_MSG"
    STASH_CREATED=true
  else
    STASH_CREATED=false
  fi

  # 4. Create task branch
  local slug
  slug=$(echo "$task_slug" | tr '[:upper:]' '[:lower:]' \
    | tr ' ' '-' | tr -cd '[:alnum:]-' | cut -c1-40)
  TASK_BRANCH="vibe/${slug}-$(date +%H%M)"
  git checkout -b "$TASK_BRANCH"
  echo "Working branch: $TASK_BRANCH"

  # 5. Export rollback command for user
  cat <<EOF

  ╔════════════════════════════════════════════╗
  ║  Rollback command (save this):             ║
  ║  vibe undo                                 ║
  ║  or manually:                              ║
  ║  git checkout $BASE_BRANCH                 ║
  ║  git branch -D $TASK_BRANCH               ║
  $([ "$STASH_CREATED" = true ] && echo "║  git stash pop                             ║")
  ╚════════════════════════════════════════════╝
EOF
}
```

## Mid-Task: Structured Commit After Each Verified Step

```bash
git_commit_step() {
  local step_description="$1"
  local test_result="${2:-unknown}"
  local lint_result="${3:-unknown}"

  # Stage only tracked + new files (not unrelated noise)
  git add -A

  # Abort if nothing changed
  if git diff --cached --quiet; then
    echo "Nothing to commit for step: $step_description"
    return 0
  fi

  # Build structured commit message
  local stat
  stat=$(git diff --cached --stat | tail -1)

  git commit -m "vibe(step): $step_description

Stats: $stat
Tests: $test_result
Lint:  $lint_result
Branch: $(git branch --show-current)
Time: $(date +%H:%M)"

  echo "✅ Committed: $step_description"
  echo "   $(git log --oneline -1)"
}
```

## Diff Review (self-check before subagent)

```bash
git_diff_review() {
  echo "=== Changes since task start ==="
  git diff "$(git merge-base HEAD origin/main 2>/dev/null || git log --oneline | tail -1 | awk '{print $1}')" HEAD \
    --stat 2>/dev/null || git diff HEAD~1 --stat

  echo ""
  echo "=== Full diff ==="
  git diff "$(git merge-base HEAD origin/main 2>/dev/null || echo HEAD~1)" HEAD \
    2>/dev/null || git diff HEAD~1

  echo ""
  echo "=== Self-check ==="
  # Flag common issues
  git diff HEAD~1 | grep -n "TODO\|FIXME\|HACK\|XXX\|console\.log\|print(\|debugger\|pdb\.set_trace\|breakpoint()" \
    && echo "⚠️  Debug artifacts found — remove before committing" \
    || echo "✅ No debug artifacts"

  git diff HEAD~1 | grep -n "password\|secret\|token\|api_key\|private_key" \
    && echo "⚠️  Possible secrets in diff — review carefully" \
    || echo "✅ No obvious secrets"
}
```

## Post-Task: Final Commit + PR

```bash
git_commit_final() {
  local intent="$1"
  local plan_file="$2"
  local test_summary="$3"
  local cycles_used="$4"

  git add -A

  if git diff --cached --quiet; then
    echo "All changes already committed in step commits."
  else
    git commit -m "vibe: $intent

$(git diff --cached --stat | head -10)
Tests: $test_summary
Correction cycles used: $cycles_used
Plan: $plan_file"
  fi

  echo ""
  echo "=== Final commit summary ==="
  git log "$(git merge-base HEAD origin/main 2>/dev/null || echo HEAD~5)"..HEAD \
    --oneline 2>/dev/null || git log --oneline -5
}

git_push_pr() {
  local intent="$1"
  local branch
  branch=$(git branch --show-current)

  git push origin "$branch"

  # Open PR if gh CLI is available
  if command -v gh &>/dev/null; then
    gh pr create \
      --title "vibe: $intent" \
      --body "Auto-generated by Hermes vibe-coding skill.

## Summary
$intent

## Changes
$(git diff main..HEAD --stat 2>/dev/null | head -10)

## Verification
- Tests: PASSED
- Lint: CLEAN
- Independent review: PASSED" \
      --draft
    echo "Draft PR created."
  else
    echo "Push complete. Create PR at:"
    echo "  https://github.com/$(git remote get-url origin | sed 's/.*github.com[:/]//' | sed 's/.git$//')/pull/new/$branch"
  fi
}
```

## Rollback

```bash
git_rollback() {
  local task_branch
  task_branch=$(git branch --show-current)

  if [[ "$task_branch" != vibe/* ]]; then
    echo "Not on a vibe branch (current: $task_branch). Nothing to undo."
    return 1
  fi

  local base_branch="${BASE_BRANCH:-main}"

  echo "Rolling back: $task_branch → $base_branch"
  git checkout "$base_branch"
  git branch -D "$task_branch"

  if [[ "$STASH_CREATED" == true ]]; then
    git stash pop && echo "Stash restored." || echo "Could not restore stash."
  fi

  echo "✅ Rollback complete. Back on $base_branch."
}
```
